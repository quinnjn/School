\begin{question}{5}
Let $\epsilon$ : $\Z+ \rightarrow \Z^+$ be the Euler-phi function
\end{question}


\begin{question}{5a}
\begin{align*}
$\phi(p^4q^2)$ =& $\phi(p^4)\phi(q^2)$\\
=&$ (p^{4-1}(p-1))(q^{2-1}(q-1))$\\
=&$ (p^{3}(p-1))(q(q-1))$\\
\end{align*}
\end{question}

\begin{question}{5b}
\begin{align*}

$\phi(343000)$ 
=& $\phi(343)\phi(1000)$\\
=& $\phi(7^3)\phi(2^3)\phi(5^3)$\\
=& $7^2(7-1)) (2^2(2-1)) (5^2(5-1)$\\
=& $7^2(6))*(2^2)*(5^2(4)$\\
=& $49(6))*4*(25(4)$\\
=& $294*4*100$\\
=& $117,600$

\end{align*}
\end{question}

\begin{question}{5c}
\begin{align*}

$\phi(n)$ 
=& $n^{1-1}(n-1)$\\
=& $n^{0}(n-1)$\\
=& $(n-1)$
\end{align*}

$\phi(101)=100$
\end{question}

\begin{question}{5d}

$\phi(2m)$
=& $\phi(2) \phi(m)$\\
=& $2^{1-1}(2-1) \phi(m)$\\
=& $(2-1) \phi(m)$\\
=& $1\phi(m)$\\
=& $\phi(m)$\\

\end{question}
